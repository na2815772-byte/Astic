document.getElementById('loginForm').addEventListener('submit', function(e) { e.preventDefault(); const loginIdentifier = document.getElementById('loginIdentifier').value; const loginPassword = document.getElementById('loginPassword').value; // LocalStorage থেকে রেজিস্টার্ড ইউজারের ডেটা আনা হচ্ছে const savedData = localStorage.getItem('registeredUser'); if (savedData) { const user = JSON.parse(savedData); // ইমেইল অথবা ফোন নম্বর এবং পাসওয়ার্ড চেক করা হচ্ছে if ((loginIdentifier === user.email || loginIdentifier === user.phone) && loginPassword === user.password) { alert(`স্বাগতম, ${user.fullName}! লগইন সফল হয়েছে।`); // সফলভাবে লগইন হলে ইউজারকে মূল হোমে (index.html) ফেরত পাঠানো হচ্ছে window.location.href = 'index.html'; } else { alert('ভুল ইমেইল/ফোন নম্বর অথবা পাসওয়ার্ড! আবার চেষ্টা করুন।'); } } else { alert('কোনো অ্যাকাউন্ট খুঁজে পাওয়া যায়নি! প্রথমে একটি অ্যাকাউন্ট তৈরি করুন।'); window.location.href = 'registry.html'; } });
